```python
def get_mapping_by_uuid(self, mapping_uuid: str) -> Optional[Dict]:
    """Get a mapping by its stable UUID"""
    conn = self.db.get_connection()
    try:
        cursor = conn.execute(
            """
            SELECT id, ke_id, ke_title, wp_id, wp_title, connection_type,
                   confidence_level, created_by, created_at, updated_at,
                   uuid, approved_by_curator, approved_at_curator, updated_by
            FROM mappings
            WHERE uuid = ?
            """,
            (mapping_uuid,)
        )
        row = cursor.fetchone()
        return dict(row) if row else None
    finally:
        conn.close()
```

@main_bp.route("/mappings/<string:mapping_uuid>")
def mapping_detail(mapping_uuid):
    """Stable mapping detail page — accessible via permanent UUID URL."""
    mapping = mapping_model.get_mapping_by_uuid(mapping_uuid)
    if not mapping:
        abort(404)
    return render_template("mapping_detail.html", mapping=mapping)
