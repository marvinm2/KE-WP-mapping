**`generate_ke_wp_gmt(mappings, cache_model=None, min_confidence=None) -> str`**
- `mappings`: list of dicts from `MappingModel.get_all_mappings()` — each has `ke_id`, `ke_title`, `wp_id`, `wp_title`, `confidence_level`
- `cache_model`: optional `CacheModel` instance for SPARQL caching (same instance used by v1_api.py); pass None to skip cache
- `min_confidence`: optional string e.g. `"high"` — filter rows where `row["confidence_level"].lower() == min_confidence`
- Returns GMT content as a string (empty string if no rows survive filtering or no genes found)

Term name format (LOCKED by user): `KE{N}_{Title_Slug}_{WP_ID}` — one row per KE-pathway pair.
- Extract numeric part: `num = re.sub(r'\\D', '', ke_id)` (e.g. "KE 55" → "55")
- Slug the KE title: `unicodedata.normalize("NFKD", ke_title).encode("ascii","ignore").decode("ascii")` then `re.sub(r'[^a-zA-Z0-9]+', '_', title).strip('_')`
- Final: `f"KE{num}_{title_slug}_{wp_id}"` e.g. `KE55_Decreased_BDNF_WP4655`

Description field (col 2): use `wp_title` — the WikiPathways pathway name.

Gene columns: HGNC symbols from WikiPathways SPARQL `wp:bdbHgncSymbol`. Use `_fetch_pathway_genes_batch(wp_ids, cache_model)` to batch all unique WP IDs in one SPARQL VALUES query:

```python
WIKIPATHWAYS_SPARQL = "https://sparql.wikipathways.org/sparql"

def _fetch_pathway_genes_batch(wp_ids: list, cache_model=None) -> dict:
    """Returns {wp_id: [hgnc_symbol, ...]} for all given wp_ids. Skips failures silently."""
    if not wp_ids:
        return {}
    import hashlib, json, requests
    values_clause = " ".join(
        [f'<https://identifiers.org/wikipathways/{wid}>' for wid in wp_ids]
    )
    query = f"""
PREFIX wp: <http://vocabularies.wikipathways.org/wp#>
PREFIX dcterms: <http://purl.org/dc/terms/>
SELECT DISTINCT ?pathwayID ?geneSymbol WHERE {{
  ?pathway a wp:Pathway ;
           dcterms:identifier ?pathwayID .
  ?geneProduct dcterms:isPartOf ?pathway ;
               wp:bdbHgncSymbol ?geneSymbol .
  VALUES ?pathway {{ {values_clause} }}
}}
"""
    query_hash = hashlib.md5(query.encode()).hexdigest()
    # Try cache first
    if cache_model:
        cached = cache_model.get_cached_response(WIKIPATHWAYS_SPARQL, query_hash)
        if cached:
            data = json.loads(cached)
            return _parse_gene_bindings(data)
    try:
        resp = requests.post(
            WIKIPATHWAYS_SPARQL,
            data={"query": query},
            headers={"Accept": "application/sparql-results+json"},
            timeout=60,
        )
        resp.raise_for_status()
        data = resp.json()
        if cache_model:
            cache_model.cache_response(WIKIPATHWAYS_SPARQL, query_hash, resp.text, expiry_hours=24)
        return _parse_gene_bindings(data)
    except Exception as e:
        logger.warning("WikiPathways SPARQL batch gene fetch failed: %s", e)
        return {}

def _parse_gene_bindings(data: dict) -> dict:
    result = {}
    for binding in data.get("results", {}).get("bindings", []):
        pid = binding.get("pathwayID", {}).get("value", "")
        gene = binding.get("geneSymbol", {}).get("value", "")
        if pid and gene:
            result.setdefault(pid, []).append(gene)
    return result
```

Namespaces:
```python
from rdflib import Graph, URIRef, Literal, RDF, Namespace
from rdflib.namespace import DCTERMS, XSD

KEWP = Namespace("https://ke-wp-mapping.org/vocab#")
MAPPING = Namespace("https://ke-wp-mapping.org/mapping/")
```

For each mapping row (after confidence filter, skip rows without `uuid`):
```python
uri = MAPPING[row["uuid"]]
g.add((uri, RDF.type, KEWP.KeyEventPathwayMapping))
g.add((uri, DCTERMS.identifier, Literal(row["uuid"])))
g.add((uri, KEWP.keyEventId, Literal(row["ke_id"])))
g.add((uri, KEWP.keyEventName, Literal(row["ke_title"])))
g.add((uri, KEWP.pathwayId, Literal(row["wp_id"])))
g.add((uri, KEWP.pathwayTitle, Literal(row["wp_title"])))
g.add((uri, KEWP.confidenceLevel, Literal(row["confidence_level"])))
if row.get("approved_by_curator"):
    g.add((uri, DCTERMS.creator, Literal(row["approved_by_curator"])))
if row.get("approved_at_curator"):
    g.add((uri, DCTERMS.date, Literal(row["approved_at_curator"], datatype=XSD.dateTime)))
if row.get("suggestion_score") is not None:
    g.add((uri, KEWP.suggestionScore, Literal(float(row["suggestion_score"]), datatype=XSD.decimal)))
```

Bind prefixes: `g.bind("ke-wp", KEWP)`, `g.bind("dcterms", DCTERMS)`, `g.bind("mapping", MAPPING)`.

Return `g.serialize(format="turtle")` — this returns a str in rdflib >= 6.0 (rdflib 6.3.2 is installed).
