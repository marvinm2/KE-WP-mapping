```python
def get_mapping_stats():
    """
    Compute aggregate mapping statistics from both mapping tables.
    Returns a dict with wp_total, go_total, total, wp_by_confidence, go_by_confidence.
    """
    stats = {
        "wp_total": 0,
        "go_total": 0,
        "total": 0,
        "wp_by_confidence": {},
        "go_by_confidence": {},
    }
    try:
        with mapping_model.db.get_connection() as conn:
            stats["wp_total"] = conn.execute("SELECT COUNT(*) FROM mappings").fetchone()[0]
            for row in conn.execute(
                "SELECT LOWER(confidence_level), COUNT(*) FROM mappings GROUP BY LOWER(confidence_level)"
            ).fetchall():
                stats["wp_by_confidence"][row[0]] = row[1]
    except Exception as e:
        logger.warning("Failed to query mapping stats: %s", e)
    try:
        if go_mapping_model:
            with go_mapping_model.db.get_connection() as conn:
                stats["go_total"] = conn.execute("SELECT COUNT(*) FROM ke_go_mappings").fetchone()[0]
                for row in conn.execute(
                    "SELECT LOWER(confidence_level), COUNT(*) FROM ke_go_mappings GROUP BY LOWER(confidence_level)"
                ).fetchall():
                    stats["go_by_confidence"][row[0]] = row[1]
    except Exception as e:
        logger.warning("Failed to query GO mapping stats: %s", e)
    stats["total"] = stats["wp_total"] + stats["go_total"]
    return stats
```

```python
def _respond_collection(serialized_rows, pagination, csv_fields):
    """
    Return JSON or CSV based on Accept header or ?format=csv query param.
    JSON: {"data": [...], "pagination": {...}}
    CSV:  header row + data rows (provenance flattened)
    """
    format_param = request.args.get("format", "").lower()
    if format_param == "csv":
        use_csv = True
    else:
        best = request.accept_mimetypes.best_match(
            ["application/json", "text/csv"], default="application/json"
        )
        use_csv = best == "text/csv"

    if use_csv:
        flat_rows = [_flatten_for_csv(r) for r in serialized_rows]
        output = io.StringIO()
        writer = csv.DictWriter(
            output, fieldnames=csv_fields, extrasaction="ignore"
        )
        writer.writeheader()
        writer.writerows(flat_rows)
        output.seek(0)
        response = make_response(output.getvalue())
        response.headers["Content-Type"] = "text/csv; charset=utf-8"
        response.headers["Content-Disposition"] = "attachment; filename=ke_wp_mappings.csv"
        return response
    return jsonify({"data": serialized_rows, "pagination": pagination})
```

This preserves full backward compatibility — existing `Accept: text/csv` header still works, and the new `?format=csv` param also works.
