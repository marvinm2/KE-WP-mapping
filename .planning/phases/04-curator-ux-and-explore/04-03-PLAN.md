// Filter state
const wpState = { aopId: null, aopLabel: null, confidence: null };

const wpTable = $('#datasetTable').DataTable({
    serverSide: true,
    processing: true,
    ajax: function(data, callback) {
        const params = new URLSearchParams({
            page: Math.floor(data.start / data.length) + 1,
            per_page: data.length,
        });
        if (wpState.aopId) params.set('aop_id', wpState.aopId);
        if (wpState.confidence) params.set('confidence_level', wpState.confidence);
        fetch(`/api/v1/mappings?${params}`, { headers: { 'Accept': 'application/json' } })
            .then(r => r.json())
            .then(json => callback({
                draw: data.draw,
                recordsTotal: json.pagination ? json.pagination.total : 0,
                recordsFiltered: json.pagination ? json.pagination.total : 0,
                data: json.data || []
            }))
            .catch(() => callback({ draw: data.draw, recordsTotal: 0, recordsFiltered: 0, data: [] }));
    },
    columns: [
        { data: 'ke_id' },
        { data: 'ke_name' },

**AOP selection handler for gaps tab:**
```javascript
$('#gaps-aop-select').on('change', function() {
    const aopId = $(this).val();
    if (!aopId) {
        $('#gaps-placeholder').show();
        $('#gaps-content').hide();
        return;
    }
    $('#gaps-placeholder').hide();
    $('#gaps-content').show();
    $('#gaps-summary').text('Loading…');
    loadCoverageGaps(aopId);
});

function loadCoverageGaps(aopId) {
    // Step 1: get all KEs in the AOP
    const encodedAopId = encodeURIComponent(aopId);
    $.getJSON(`/get_aop_kes/${encodedAopId}`)
        .done(function(aopKes) {
            // aopKes is a list of objects with ke_id, ke_title, biolevel (or similar)
            // Step 2: get mapped KE IDs for this AOP from v1 API
            $.getJSON(`/api/v1/mappings?aop_id=${encodedAopId}&per_page=200`)
                .done(function(response) {
                    const mappedKeIds = new Set((response.data || []).map(m => m.ke_id));
                    const gapKes = aopKes.filter(ke => {
                        const keId = ke.ke_id || ke.keId || ke.KElabel || '';
                        return !mappedKeIds.has(keId);
                    });
                    renderGapsTable(gapKes, aopKes.length, mappedKeIds.size);
                })
                .fail(function() {
                    $('#gaps-summary').text('Error loading mapped KEs. Please try again.');
                });
        })
        .fail(function() {
            $('#gaps-summary').text('Error loading KEs for this AOP. Please try again.');
        });
}

function renderGapsTable(gapKes, totalKes, mappedCount) {
    const gapCount = gapKes.length;
    $('#gaps-summary').html(
        `<strong>${mappedCount}</strong> of <strong>${totalKes}</strong> KEs in this AOP have approved KE-WP mappings. ` +
        `<strong style="color:${gapCount > 0 ? '#c62828' : '#2e7d32'};">${gapCount} KE${gapCount !== 1 ? 's' : ''}</strong> ` +
        `${gapCount > 0 ? 'have no mapping yet.' : 'all mapped!'}`
    );

    if (gapCount === 0) {
        $('#gapsTable').hide();
        $('#gaps-all-mapped').show();
        return;
    }

    $('#gaps-all-mapped').hide();

    // Destroy existing DataTable if initialized
    if ($.fn.DataTable.isDataTable('#gapsTable')) {
        $('#gapsTable').DataTable().destroy();
        $('#gapsTable tbody').empty();
    }

    // Populate rows
    gapKes.forEach(function(ke) {
        const keId = ke.ke_id || ke.keId || ke.KElabel || '';
        const keTitle = ke.ke_title || ke.keTitle || ke.KEtitle || '';
        const biolevel = ke.biolevel || '';
        const encodedKe = encodeURIComponent(keId);
        $('#gapsTable tbody').append(`<tr>
            <td>${keId}</td>
            <td>${keTitle}</td>
            <td>${biolevel}</td>
            <td><a href="/?ke_id=${encodedKe}" class="btn btn-sm" style="padding:4px 10px;background:#307BBF;color:white;border-radius:4px;text-decoration:none;font-size:13px;">Map</a></td>
        </tr>`);
    });

    $('#gapsTable').show().DataTable({
        order: [[0, 'asc']],
        pageLength: 25,
        columnDefs: [
            { width: '10%', targets: 0, className: 'dt-center' },
            { width: '50%', targets: 1 },
            { width: '20%', targets: 2, className: 'dt-center' },
            { width: '20%', targets: 3, className: 'dt-center' }
        ]
    });
}
```
